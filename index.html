#My Piet Mondrian-influenced twitter cover photo, feel free to copy & experiment 
import random
import matplotlib.pyplot as plt
import numpy as np

fig = plt.figure(figsize=(20,10))
plt.axis('off')
plt.xlim(-1,1)
plt.ylim(-1,1)
num_lines = 2000

for i in range(num_lines):
    
    y=np.random.normal(0,0.3)
    x1=np.random.normal(0,0.35)
    x2=x1+random.uniform(0.01,0.025) 
    color = '#%02x%02x%02x' % (random.randint(10,20),random.randint(10,20),random.randint(10,20))
    plt.hlines(y, x1, x2, color=color,linewidth=random.uniform(0,2), alpha=random.uniform(0.5,1))
 
    x=np.random.normal(0,0.3)
    y1=np.random.normal(0,0.35)
    y2=y1+random.uniform(0.01,0.025) 
    color = '#%02x%02x%02x' % (random.randint(10,20),random.randint(10,20),random.randint(10,20))
    plt.vlines(x,y1,y2, color=color,linewidth=random.uniform(0,2), alpha=random.uniform(0.5,1))

plt.show()
